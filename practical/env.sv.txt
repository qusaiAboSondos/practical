class env;

   agent agt;
   monitor mon;
   scoreboard sb;

   mailbox #(packet) gen2drv;
   mailbox #(packet) gen2sb;
   mailbox #(result_pkt) mon2sb;

   function new(
      virtual pkt_if.DRIVER drv_vif,
      virtual pkt_if.MONITOR mon_vif
   );

      gen2drv = new();
      gen2sb  = new();
      mon2sb  = new();

      agt = new(
         drv_vif,
         gen2drv,
         gen2sb
      );

      mon = new(
         mon_vif,
         mon2sb
      );

      sb = new(
         gen2sb,
         mon2sb
      );

   endfunction

   task run();

      fork

         agt.drv.run();
         mon.run();
         sb.run();

      join_none

      agt.gen.run(50);

   endtask

endclass
